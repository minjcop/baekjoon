// a[n][k] : n번째 집을 k색깔로 칠하는 비용(k = 0 빨강, k = 1 초록, k = 2 파랑)
// dp[n][k] : n번째까지 칠하는 비용의 최솟값, k = 마지막색깔.
// min(dp[n][0], dp[n][1], dp[n][2]) 가 정답임.

// 점화식 : 
dp[n][0] = a[n][0] + min(dp[n-1][1], dp[n-1][2]);
dp[n][1] = a[n][1] + min(dp[n-1][0], dp[n-1][2]);
dp[n][2] = a[n][2] + min(dp[n-1][0], dp[n-1][1]);
